```ts
function AsyncComponent(props: { func: () => Promise<JSX.Element> }) {
  const [loaded, setLoaded] = useState<JSX.Element>(null);
  useEffect(() => {
    props.func().then(result => {
      setLoaded(result);
    });
  }, []);
  return loaded ? loaded : <div>Loading...</div>
}
```
